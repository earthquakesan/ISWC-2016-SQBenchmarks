SELECT  ?entity ?label1
WHERE
  { OPTIONAL
      { ?entity <http://www.w3.org/2000/01/rdf-schema#label> ?label1
        FILTER regex(str(?label1), "amazon", "i")
      }
    FILTER bound(?label1)
    FILTER isIRI(?entity)
  }
LIMIT   40

PREFIX  ical: <http://www.w3.org/2002/12/cal/ical#>
PREFIX  dct:  <http://purl.org/dc/terms/>
PREFIX  skos: <http://www.w3.org/2004/02/skos/core#>
PREFIX  rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX  foaf: <http://xmlns.com/foaf/0.1/>
PREFIX  dc:   <http://purl.org/dc/elements/1.1/>

SELECT DISTINCT  ?s ?type ?name ?prefLabel
WHERE
  { ?s <http://www.w3.org/1999/02/22-rdf-syntax-ns#type> ?type
    OPTIONAL
      {   { ?s foaf:name ?name }
        UNION
          { ?s dc:title ?name }
        UNION
          { ?s dct:title ?name }
        UNION
          { ?s ical:summary ?name }
        UNION
          { ?s rdfs:label ?name }
      }
    OPTIONAL
      { ?s skos:prefLabel ?prefLabel }
    FILTER regex(str(?s), "^http://data.semanticweb.org")
    FILTER ( ! regex(str(?s), "^http://data.semanticweb.org/ns") )
  }
#-------------------------------------------------------
PREFIX  rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX  swrc: <http://swrc.ontoware.org/ontology#>
PREFIX  foaf: <http://xmlns.com/foaf/0.1/>
PREFIX  swc:  <http://data.semanticweb.org/ns/swc/ontology#>
PREFIX  dc:   <http://purl.org/dc/elements/1.1/>

SELECT DISTINCT  ?abstract ?keyword ?author_name
WHERE
  {   { <http://data.semanticweb.org/conference/eswc/2014/paper/demo/38> swrc:author ?author }
    UNION
      { <http://data.semanticweb.org/conference/eswc/2014/paper/demo/38> foaf:maker ?author }
    ?author foaf:name ?author_name
    OPTIONAL
      { <http://data.semanticweb.org/conference/eswc/2014/paper/demo/38> swrc:abstract ?abstract }
    OPTIONAL
      {   { <http://data.semanticweb.org/conference/eswc/2014/paper/demo/38> swrc:keywords ?keyword }
        UNION
          { <http://data.semanticweb.org/conference/eswc/2014/paper/demo/38> dc:subject ?keyword }
        UNION
          { <http://data.semanticweb.org/conference/eswc/2014/paper/demo/38> swc:hasTopic ?topic .
            ?topic rdfs:label ?keyword
          }
      }
  }
#-------------------------------------------------------
PREFIX  geo:  <http://www.w3.org/2003/01/geo/wgs84_pos#>
PREFIX  swrc: <http://swrc.ontoware.org/ontology#>
PREFIX  ical: <http://www.w3.org/2002/12/cal/ical#>
PREFIX  rdf:  <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX  owl:  <http://www.w3.org/2002/07/owl#>
PREFIX  swrc-ext: <http://www.cs.vu.nl/~mcaklein/onto/swrc_ext/2005/05#>
PREFIX  rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX  foaf: <http://xmlns.com/foaf/0.1/>
PREFIX  swc:  <http://data.semanticweb.org/ns/swc/ontology#>
PREFIX  dc:   <http://purl.org/dc/elements/1.1/>

SELECT DISTINCT  ?s ?t ?y ?to ?h
WHERE
  { ?s dc:title ?t .
    ?s swrc:year ?y
    OPTIONAL
      { ?s foaf:homepage ?h }
    OPTIONAL
      { ?s foaf:topic ?to }
  }
ORDER BY DESC(?y)
LIMIT   10
#-------------------------------------------------------
PREFIX  skos: <http://www.w3.org/2004/02/skos/core#>
PREFIX  bibo: <http://purl.org/ontology/bibo/>
PREFIX  rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX  foaf: <http://xmlns.com/foaf/0.1/>
PREFIX  swrcext: <http://www.cs.vu.nl/~mcaklein/onto/swrc_ext/2005/05#>

SELECT DISTINCT  ?pred ?author_url ?author_name ?author_pref_label
WHERE
  { GRAPH ?graph
      {   { <http://data.semanticweb.org/conference/iswc/2010/paper/274> bibo:authorList ?authorList }
        UNION
          { <http://data.semanticweb.org/conference/iswc/2010/paper/274> swrcext:authorList ?authorList }
        ?authorList ?pred ?author_url
          { ?author_url foaf:name ?author_name }
        UNION
          { ?author_url rdfs:label ?author_name }
        OPTIONAL
          { ?author_url skos:prefLabel ?author_pref_label }
      }
  }
ORDER BY ?pred
